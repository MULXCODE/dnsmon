define(["d3"], function(){

    var PTimeView = function(inputVector){
        var domainRange, grainLevelsDefault, pixelsRange, boardHeight, picker, xAxis, svgContainer, svgPicker,
            currentSelection, actualGrain, grainLevels, boardWidth;

        this.id = inputVector.id;
        this.dom = {};
        this.dom.node = this.dom.$.node;

        this.domain = {};
        this.domain.start = initVector.domainStart;
        this.domain.stop = initVector.domainStop;
        this.domain.duration = this.domain.stop - this.domain.start;
        this.margin = {top: inputVector.top, right: inputVector.right, bottom: inputVector.bottom, left: inputVector.left};
        this.width = inputVector.width;
        this.height = inputVector.height;
        this.alignment = (inputVector.alignment == "horizontal") ? "horizontal" : "vertical";
        this.labelsOrientation = inputVector.labelsOrientation || "-65";

        domainRange = [this.domain.start, this.domain.stop];

        this.callBacks = {"onRelease": [], "onDrag": [], "click": [], "onChange": []};

        /*
         * Default granularity
         */
        grainLevelsDefault = {
            "hour": 2880 * 60 * 1000, //2 days
            "day": 43200 * 0.5 * 60 * 1000, //0.5 month
            "week": 43200 * 5 * 60 * 1000, //5 months
            "month": 43200 * 12 * 60 * 1000 //1 year
        };

        grainLevels = inputVector || grainLevelsDefault;

        this.on = function(eventName, callback, context){ //Publish/subscribe registration implementation
            var eventList;

            eventList = this.callBacks[eventName];

            if (eventList){
                eventList.push({callback: callback, context: context});
            }else{
                console.log("Not available event with this name.");
            }
        };

        this.trigger = function(eventName){ //Publish/subscribe publisher implementation
            var eventList, callbackTmp;
            eventList = this.callBacks[eventName];

            for (var n=0,length=eventList.length; n<length; n++){
                callbackTmp = eventList[n];
                callbackTmp.callback.call(callbackTmp.callback.context);
            }
        };

        this._computeGrain = function(){
            var grain;

            if (this.domain.duration < (grainLevels.hour)){
                grain = {
                    "timeMapper": d3.time.hour,
                    "timeTicker": d3.time.hours,
                    "timeGrid": d3.time.minutes,
                    "timeUnitGrid": d3.time.minute
                };
            }else if (this.domain.duration < (grainLevels.day)){
                grain = {
                    "timeMapper": d3.time.day,
                    "timeTicker": d3.time.days,
                    "timeGrid": d3.time.hours,
                    "timeUnitGrid": d3.time.hour
                };
            }else if (this.domain.duration< (grainLevels.week)){
                grain = {
                    "timeMapper": d3.time.week,
                    "timeTicker": d3.time.weeks,
                    "timeGrid": d3.time.days,
                    "timeUnitGrid": d3.time.day
                };
            }else if (this.domain.duration < (grainLevels.month)){
                grain = {
                    "timeMapper": d3.time.month,
                    "timeTicker": d3.time.months,
                    "timeGrid": d3.time.weeks,
                    "timeUnitGrid": d3.time.week
                };
            }else{
                grain = {
                    "timeMapper": d3.time.year,
                    "timeTicker": d3.time.years,
                    "timeGrid": d3.time.months,
                    "timeUnitGrid": d3.time.month
                };
            }

            return grain;
        };

        this._onRelease = function() {
            var extent0, extent1;

            if (!d3.event.sourceEvent) return;

            extent0 = picker.extent();
            extent1 = extent0.map(actualGrain.timeUnitGrid.round);

            // if empty when rounded, use floor & ceil instead
            if (extent1[0] >= extent1[1]) {
                extent1[0] = actualGrain.timeUnitGrid.floor(extent0[0]);
                extent1[1] = actualGrain.timeUnitGrid.ceil(extent0[1]);
            }

            d3.select(this)
                .transition()
                .call(picker.extent(extent1));

            this.selection.start = extent1[0];
            this.selection.stop = extent1[1];

            this.trigger("onRelease");
        };

        this._onDrag = function() {
            var extent0, extent1;
            if (!d3.event.sourceEvent) return;

            extent0 = picker.extent();
            extent1 = extent0.map(actualGrain.timeUnitGrid.round);

            if (extent1[0] >= extent1[1]) {
                extent1[0] = actualGrain.timeUnitGrid.floor(extent0[0]);
                extent1[1] = actualGrain.timeUnitGrid.ceil(extent0[1]);
            }

            d3.select(this)
                .transition()
                .call(picker.extent(extent1));

            this.trigger("onDrag");
        };


        this.render = function(renderVector){
            this.selection = {};
            this.selection.start = renderVector.selectionStart;
            this.selection.stop = renderVector.selectionStop;

            currentSelection = [this.selection.start, this.selection.stop];
            actualGrain = this._computeGrain();
            pixelsRange = [0, this.width];
            boardWidth = this.width + this.margin.left + this.margin.right;
            boardHeight = this.height + this.margin.top + this.margin.bottom;

            xAxis = d3.time.scale()
                .domain(domainRange)
                .range(pixelsRange);

            picker = d3.svg.brush()
                .x(xAxis)
                .extent(currentSelection)
                .on("brush", this._onDrag)
                .on("brushend", this._onRelease);

            svgContainer = d3.select(this.dom.node)
                .append("svg")
                .attr("class", "brusher")
                .attr("width", boardWidth)
                .attr("height", boardHeight)
                .append("g")
                .attr("transform", "translate(" + this.margin.left + "," + this.margin.top + ")");

            svgContainer.append("rect")
                .attr("class", "grid-background")
                .attr("width", this.width)
                .attr("height", this.height);

            svgContainer.append("g")
                .attr("class", "x grid")
                .attr("transform", "translate(0," + this.height + ")")
                .call(
                    d3.svg
                        .axis()
                        .scale(xAxis)
                        .orient("bottom")
                        .ticks(actualGrain.timeGrid)
                        .tickSize(- this.height)
                        .tickFormat(""))
                .selectAll(".tick")
                .classed("minor", function(d) { return d.getHours(); });

            svgContainer.append("g")
                .attr("class", "x axis")
                .attr("transform", "translate(0," + this.height + ")")
                .call(
                    d3.svg
                        .axis()
                        .scale(xAxis)
                        .orient("bottom")
                        .ticks(actualGrain.timeTicker)
                        .tickFormat(d3.time.format("%d-%m-%Y"))
                        .tickPadding(0))
                .selectAll("text")
                .attr("x", 6)
                .style("text-anchor", null);

            svgPicker = svgContainer.append("g")
                .attr("class", "brush")
                .call(picker);

            svgPicker.selectAll("rect")
                .attr("height", this.height);

            svgContainer.selectAll("text")
                .style("text-anchor", "end")
                .attr("dx", "-1.2em")
                .attr("dy", ".15em")
                .attr('transform', 'rotate(' + this.labelsOrientation + ')');
        };

        this.update = function(updateVector){
            var renderVector;

            if (this.selection.start != updateVector.selectionStart || this.selection.stop != updateVector.selectionStop){

                this.selection.start = updateVector.selectionStart || this.selection.start;
                this.selection.stop = updateVector.selectionStop || this.selection.stop;

                if (this.domain.start != updateVector.domainStart || this.domain.stop != updateVector.domainStop){
                    this.domain.start = updateVector.domainStart;
                    this.domain.stop = updateVector.domainStop;

                    renderVector = {
                        "selectionStart": this.selection.start,
                        "selectionStop": this.selection.stop
                    };

                    d3.select(this.dom.node).remove();
                    this.render(renderVector);

                }else{

                    currentSelection = [this.selection.start, this.selection.stop];

                    picker = d3.svg
                        .brush()
                        .x(xAxis)
                        .extent(currentSelection);

                    this.trigger("onChange");
                }
            }
        };
    };

});
